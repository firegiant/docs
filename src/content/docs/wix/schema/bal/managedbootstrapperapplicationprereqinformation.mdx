Adds license information for a prereq package, should only be used when unable to add the license attributes to the package directly.

 _Deprecated in WiX v5; use BootstrapperApplicationPrerequisiteInformation instead._

## Attributes

<BlockWrapper>

  <BlockReference
    content={{
      title: "LicenseFile",
      type: "string"
    }}
  >
  Source file of the license. May not be used with LicenseUrl.
  </BlockReference>

  <BlockReference
    content={{
      title: "LicenseUrl",
      type: "string"
    }}
  >
  Source url of the license. May not be used with LicenseFile.
  </BlockReference>

  <BlockReference
    content={{
      title: "PackageId",
      type: "string",
      required: true
    }}
  >
  Id of the target package.
  </BlockReference>

</BlockWrapper>
